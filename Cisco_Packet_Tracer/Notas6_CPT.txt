	conf t
	no ip domain-lookup
	banner motd #Cuidao con lo que hace manito#
	hostname NOMBRE_DEL_DISPOSITIVO
	enable secret klk001#(CONTRASEÑA)
	username JotasCode(NOMBRE_DEL_USUARIO) privilege 15 secret hola123@(CONTRASEÑA)
	ip domain-name DOMINIO.COM
	crypto key generate rsa
	1024
	ip ssh time-out 60
	ip ssh version 2
	ip ssh authentication-retries 3
	line con 0
	pass nba123
	logging synchronous 
	login local 
	line vty 0 15
	logging synchronous 
	transport input ssh
	no transport input telnet
	end
	copy running-config startup-config
	reload
